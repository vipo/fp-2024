<round_command> ::= "roundTo " <product>

<check_shipping_command> ::= "checkShipping " <product>

<add_command> ::= "add " <product> | "add " <products>

<discount_command> ::= "giveDiscount " <product> " " <discount>

<buy_command> ::= "buy " <quantity> " " <product>

<compare_command> ::= "compare " <product> " " <product>


<products> ::= <product> | <product> ", " <products>
<product> ::= <boardgame_with_addons> | <boardgame> | <add_on> | <component>


<boardgame> ::=  <boardgame_name> " " <price>  " (contains: " <products> ")"

<boardgame_with_addons> ::= <boardgame> "[includes: " <products> "]"

<component> ::= <quantity> " " <component_name>

<add_on> ::= <add_on_name> " " <price>


<quantity> ::= <number>

<discount> ::= <number> "%"

<price> ::= <number> "eur" | <number> "." <number> "eur"


<number> ::= <digit> | <digit> <number>

<digit> ::= [0-9]

<boardgame_name> ::= "corporateCEOTM"
            | "baseTM"
            | "bigBoxTM"
            | "venusTMexp"
            | "turmoilTMexp"
            | "preludeTMexp"
            | "prelude1TMexp"
            | "prelude2TMexp"
            | "coloniesTMexp"
            | "ellas&hellasTMexp"
            | "automaTMexp"
            | "baseTMAE"
            | "discoveryTMAEexp"
            | "foundationsTMAEexp"
            | "crisisTMAEexp"


<component_name> ::= "tile" 
            | "gameBoard"
            | "playerBoard" 
            | "card"
            | "marker"
            | "rules"
            

<add_on_name> ::= "playerBoard" 
            | "miniature"
            | "metalResource"
            | "cardSleeve"
            | "spaceInsert"
